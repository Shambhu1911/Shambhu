//q2 import pandas as pd from sklearn.preprocessing import LabelEncoder data = pd.read_csv('Data.csv') data_with_onehot = pd.get_dummies(data, columns=['Country'], prefix=['Country']) label_encoder = LabelEncoder() data_with_onehot['Purchased'] = label_encoder.fit_transform(data['Purchased']) data_with_onehot.to_csv('Modified_Data.csv', index=False) print("OneHot encoding and Label encoding applied. Data saved to Modified_Data.csv")
